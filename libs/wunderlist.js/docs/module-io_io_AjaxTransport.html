Ajax transport that permits a maximum number of open http connections. Queueing and flushing is handled automagically.

Source:
  • io/io/AjaxTransport.js, line 3

Example

var ajax = require('io/io/AjaxTransport').ajax;
var request = ajax('http://www.example.comt/api/endpoint', {
  'type': 'GET',
  'data': {
    'param': 'value'
  },
  'success': function (data, xhr) {//...},
  'error': function (data, xhr) {//...}
});
